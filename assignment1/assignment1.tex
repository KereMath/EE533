\section{Problem 2.2 -- Entropy of Functions}
%============================================================

\subsection*{Problem Statement}
Let $X$ be a random variable taking on a finite number of values. Find the general inequality
relation between $\HH(X)$ and $\HH(Y)$ if:
\begin{enumerate}[label=(\alph*)]
  \item $Y = 2^X$
  \item $Y = \cos X$
\end{enumerate}

\subsection*{Solution}

\paragraph{(a) $Y = 2^X$.}

The function $f(x) = 2^x$ is \emph{strictly increasing}, hence \emph{one-to-one} (injective)
on any set of real values.  By the invariance-under-relabelling theorem (Lecture~1, slide~15),
\[
  \HH(X) = \HH(f(X)) \quad \text{for any injective } f.
\]
Therefore
\[
  \boxed{\HH(Y) = \HH(X).}
\]
Intuitively, knowing $Y = 2^X$ is exactly as informative as knowing $X$, because $2^X$
uniquely determines $X$.

\paragraph{(b) $Y = \cos X$.}

The cosine function is \emph{not} injective in general: for instance,
$\cos(\theta) = \cos(-\theta)$ for all $\theta$.  Hence several values of $X$ may map
to the same value of $Y$, causing information loss.

Since $Y = g(X)$ is a deterministic function of $X$, by the data-processing principle
(conditioning cannot increase entropy):
\[
  \HH(Y) \;\le\; \HH(X).
\]
The inequality is strict unless $g$ restricted to $\mathrm{supp}(p_X)$ is injective.

\[
  \boxed{\HH(Y) \le \HH(X).}
\]

\noindent\textbf{Formal argument.}  Let $Z := (X, Y) = (X, \cos X)$.  Then
\begin{align*}
  \HH(X) &= \HH(X, Y) \quad (\text{since } Y \text{ is determined by } X)\\
         &= \HH(Y) + \HH(X \mid Y)\\
         &\ge \HH(Y),
\end{align*}
since $\HH(X \mid Y) \ge 0$, with equality iff $X$ is a function of $Y$, i.e., iff $\cos$
is injective on $\mathrm{supp}(p_X)$.

\section{Problem 2.3 -- Minimum Entropy}
%============================================================

\subsection*{Problem Statement}
What is the minimum value of $\HH(p_1, \dots, p_n) = \HH(\mathbf{p})$ as $\mathbf{p}$ ranges
over the set of all $n$-dimensional probability vectors?  Find all $\mathbf{p}$'s that achieve
this minimum.

\subsection*{Solution}

\paragraph{Claim.} The minimum entropy is $\mathbf{0}$, achieved exactly by all
\emph{degenerate} (point-mass) distributions.

\paragraph{Proof.}

By definition,
\[
  \HH(\mathbf{p}) = -\sum_{i=1}^{n} p_i \log p_i \quad (\text{convention: } 0\log 0 := 0).
\]

\textbf{Lower bound.}  Each term $-p_i \log p_i \ge 0$ for $p_i \in [0,1]$
(since $\log p_i \le 0$).  Therefore $\HH(\mathbf{p}) \ge 0$.

\textbf{Achievability.}  If $p_k = 1$ for some index $k$ and $p_i = 0$ for all $i \ne k$,
then
\[
  \HH(\mathbf{p}) = -1\cdot\log 1 - \sum_{i\ne k} 0 \cdot \log 0 = 0.
\]

\textbf{Uniqueness.}  Suppose $\HH(\mathbf{p}) = 0$.  Then every term
$-p_i \log p_i = 0$, which holds iff $p_i \in \{0,1\}$.  Combined with
$\sum_i p_i = 1$, exactly one $p_k = 1$ and all others equal $0$.

\paragraph{Conclusion.}
\[
  \boxed{\min_{\mathbf{p}\in\Delta_n} \HH(\mathbf{p}) = 0,}
\]
achieved precisely by the $n$ standard basis vectors
$\mathbf{e}_1, \mathbf{e}_2, \dots, \mathbf{e}_n$ (i.e., deterministic distributions).

\section{Problem 2.10 -- Entropy of a Disjoint Mixture}
%============================================================

\subsection*{Problem Statement}
Let $X_1$ and $X_2$ be discrete random variables drawn according to pmfs $p_1(\cdot)$ and
$p_2(\cdot)$ over disjoint alphabets $\mathcal{X}_1 = \{1,\dots,m\}$ and
$\mathcal{X}_2 = \{m+1,\dots,n\}$.  Define
\[
  X = \begin{cases} X_1 & \text{with probability } \alpha,\\ X_2 & \text{with probability } 1-\alpha.\end{cases}
\]
\begin{enumerate}[label=(\alph*)]
  \item Find $\HH(X)$ in terms of $\HH(X_1)$, $\HH(X_2)$, and $\alpha$.
  \item Maximize over $\alpha$ to show $2^{\HH(X)} \le 2^{\HH(X_1)} + 2^{\HH(X_2)}$,
        and interpret via the notion that $2^{\HH(X)}$ is the effective alphabet size.
\end{enumerate}

\subsection*{Solution}

\paragraph{(a) Expression for $\HH(X)$.}

Since $\mathcal{X}_1$ and $\mathcal{X}_2$ are disjoint, the pmf of $X$ is
\[
  p_X(k) = \begin{cases} \alpha\, p_1(k) & k \in \mathcal{X}_1,\\ (1-\alpha)\,p_2(k) & k \in \mathcal{X}_2.\end{cases}
\]
Splitting the entropy sum:
\begin{align*}
  \HH(X)
    &= -\sum_{k \in \mathcal{X}_1} \alpha p_1(k)\log\!\bigl(\alpha p_1(k)\bigr)
       -\sum_{k \in \mathcal{X}_2} (1-\alpha)p_2(k)\log\!\bigl((1-\alpha)p_2(k)\bigr)\\[4pt]
    &= -\sum_{k \in \mathcal{X}_1} \alpha p_1(k)\bigl[\log\alpha + \log p_1(k)\bigr]
       -\sum_{k \in \mathcal{X}_2} (1-\alpha)p_2(k)\bigl[\log(1-\alpha)+\log p_2(k)\bigr]\\[4pt]
    &= -\alpha\log\alpha\underbrace{\sum_{k}p_1(k)}_{=1}
       + \alpha\underbrace{\Bigl(-\sum_k p_1(k)\log p_1(k)\Bigr)}_{=\HH(X_1)}\\
    &\quad -\,(1-\alpha)\log(1-\alpha)\underbrace{\sum_k p_2(k)}_{=1}
       + (1-\alpha)\underbrace{\Bigl(-\sum_k p_2(k)\log p_2(k)\Bigr)}_{=\HH(X_2)}.
\end{align*}
Defining the binary entropy $\hb(\alpha) := -\alpha\log\alpha - (1-\alpha)\log(1-\alpha)$,

\[
  \boxed{\HH(X) = \hb(\alpha) + \alpha\,\HH(X_1) + (1-\alpha)\,\HH(X_2).}
\]

\paragraph{(b) Bound $2^{\HH(X)} \le 2^{\HH(X_1)} + 2^{\HH(X_2)}$.}

Let $a := 2^{\HH(X_1)}$ and $b := 2^{\HH(X_2)}$.  From part~(a),
\[
  2^{\HH(X)} = 2^{\hb(\alpha)}\cdot a^{\alpha}\cdot b^{1-\alpha}.
\]
Maximize over $\alpha \in (0,1)$.  Setting $\tfrac{d}{d\alpha}\HH(X) = 0$:
\[
  \log\!\frac{1-\alpha}{\alpha} + \log a - \log b = 0
  \;\Longrightarrow\;
  \frac{1-\alpha}{\alpha} = \frac{b}{a}
  \;\Longrightarrow\;
  \alpha^* = \frac{a}{a+b}.
\]
At $\alpha^*$:
\begin{align*}
  \hb(\alpha^*)
    &= -\frac{a}{a+b}\log\frac{a}{a+b} - \frac{b}{a+b}\log\frac{b}{a+b}\\
    &= \frac{a}{a+b}\log\frac{a+b}{a} + \frac{b}{a+b}\log\frac{a+b}{b}.
\end{align*}
Therefore
\begin{align*}
  2^{\HH(X^*)}
    &= 2^{\hb(\alpha^*)}\cdot a^{\frac{a}{a+b}}\cdot b^{\frac{b}{a+b}}\\
    &= \left(\frac{a+b}{a}\right)^{\!\frac{a}{a+b}}
       \left(\frac{a+b}{b}\right)^{\!\frac{b}{a+b}}
       a^{\frac{a}{a+b}}\,b^{\frac{b}{a+b}}\\
    &= (a+b)^{\frac{a}{a+b}}\cdot(a+b)^{\frac{b}{a+b}}\\
    &= a + b
     = 2^{\HH(X_1)} + 2^{\HH(X_2)}.
\end{align*}
Since $\alpha^*$ maximizes $\HH(X)$, for \emph{any} $\alpha$ we have
$\HH(X) \le \HH(X^*)$, and hence
\[
  \boxed{2^{\HH(X)} \le 2^{\HH(X_1)} + 2^{\HH(X_2)}.}
\]

\paragraph{Interpretation.}  The quantity $2^{\HH(X)}$ is the \emph{effective alphabet size}
(number of ``typical'' outcomes).  The bound says that the effective alphabet size of a mixture
cannot exceed the \emph{sum} of the effective alphabet sizes of the component distributions,
and the maximum is achieved when $\alpha^* = 2^{\HH(X_1)}/(2^{\HH(X_1)}+2^{\HH(X_2)})$,
i.e., the mixing weight is proportional to each component's effective size.

\section{Problem 2.39 -- Entropy and Pairwise Independence}
%============================================================

\subsection*{Problem Statement}
Let $X$, $Y$, $Z$ be three binary Bernoulli$(\tfrac{1}{2})$ random variables that are
pairwise independent; that is,
\[
  \II(X;Y) = \II(X;Z) = \II(Y;Z) = 0.
\]
\begin{enumerate}[label=(\alph*)]
  \item What is the minimum value of $\HH(X,Y,Z)$ under this constraint?
  \item Give an example achieving this minimum.
\end{enumerate}

\subsection*{Solution}

\paragraph{Setup.}
Since each variable is Bernoulli$(\tfrac{1}{2})$:
\[
  \HH(X) = \HH(Y) = \HH(Z) = 1 \text{ bit}.
\]
Pairwise independence $\II(X;Y)=0$ implies $\HH(X,Y)=\HH(X)+\HH(Y)=2$ bits (and likewise
for the other pairs).

\paragraph{(a) Minimum value of $\HH(X,Y,Z)$.}

By the chain rule and monotonicity of conditioning:
\begin{align*}
  \HH(X,Y,Z)
    &= \HH(X,Y) + \HH(Z \mid X,Y)\\
    &= 2 + \HH(Z \mid X,Y).
\end{align*}
Since $\HH(Z \mid X,Y) \ge 0$, we get $\HH(X,Y,Z) \ge 2$.

We now show the lower bound $\HH(X,Y,Z) \ge 2$ is tight by constructing a valid example.

\paragraph{(b) Achieving example: $Z = X \oplus Y$.}

Define $Z := X \oplus Y$ (XOR / modulo-2 addition).  Then $Z$ is fully determined by
$(X,Y)$, so $\HH(Z \mid X,Y) = 0$, giving
\[
  \HH(X,Y,Z) = 2 + 0 = 2 \text{ bits.}
\]

\textbf{Verification of pairwise independence.}

Take $X,Y$ i.i.d.\ Bernoulli$(\tfrac{1}{2})$ and $Z=X\oplus Y$.

\begin{itemize}
  \item \emph{$X \perp Y$}: by assumption.
  \item \emph{$X \perp Z$}: For any $(x,z) \in \{0,1\}^2$,
    \[
      \Prob(X=x, Z=z) = \Prob(X=x, Y=z\oplus x) = \tfrac{1}{2}\cdot\tfrac{1}{2} = \tfrac{1}{4},
    \]
    and $\Prob(X=x)=\Prob(Z=z)=\tfrac{1}{2}$.  Hence $\Prob(X=x,Z=z)=\Prob(X=x)\Prob(Z=z)$.
  \item \emph{$Y \perp Z$}: By symmetry, same argument.
\end{itemize}
All three pairwise independence conditions are satisfied.

\paragraph{Conclusion.}
\[
  \boxed{\min \HH(X,Y,Z) = 2 \text{ bits},}
\]
achieved by $Z = X \oplus Y$.

Note: $X$, $Y$, $Z$ are \emph{not} mutually independent (knowing $X$ and $Y$ determines $Z$),
demonstrating that pairwise independence does not imply mutual independence.

\section{Problem 3.3 -- Piece of Cake}
%============================================================

\subsection*{Problem Statement}
A cake is sliced roughly in half; the largest piece is chosen and the rest discarded.  A random
cut creates pieces in proportions
\[
  P = \begin{cases}
    \bigl(\tfrac{2}{3},\, \tfrac{1}{3}\bigr) & \text{with probability } \tfrac{3}{4},\\[4pt]
    \bigl(\tfrac{3}{5},\, \tfrac{2}{5}\bigr) & \text{with probability } \tfrac{1}{4}.
  \end{cases}
\]
How large, to first order in the exponent, is the piece of cake after $n$ cuts?

\subsection*{Solution}

\paragraph{Setup.}  At cut $i$, the surviving piece is multiplied by a random factor
\[
  M_i = \begin{cases}
    \tfrac{2}{3} & \text{with probability } \tfrac{3}{4},\\[2pt]
    \tfrac{3}{5} & \text{with probability } \tfrac{1}{4},
  \end{cases}
\]
independently across cuts.  After $n$ cuts, the piece size is
\[
  S_n = \prod_{i=1}^{n} M_i.
\]

\paragraph{Exponent via the Law of Large Numbers.}

Taking the logarithm (base 2):
\[
  \frac{1}{n}\log S_n
    = \frac{1}{n}\sum_{i=1}^{n} \log M_i
    \;\xrightarrow{\;P\;}\; \E[\log M]
    \quad \text{as } n\to\infty,
\]
by the Weak Law of Large Numbers (LLN), since $M_i$ are i.i.d.\ with $\E[|\log M_1|]<\infty$.

\paragraph{Computing $\E[\log_2 M]$.}
\begin{align*}
  \E[\log_2 M]
    &= \frac{3}{4}\log_2\!\frac{2}{3} + \frac{1}{4}\log_2\!\frac{3}{5}\\[4pt]
    &= \frac{3}{4}(1-\log_2 3) + \frac{1}{4}(\log_2 3 - \log_2 5)\\[4pt]
    &= \frac{3}{4} - \frac{3}{4}\log_2 3 + \frac{1}{4}\log_2 3 - \frac{1}{4}\log_2 5\\[4pt]
    &= \frac{3}{4} - \frac{1}{2}\log_2 3 - \frac{1}{4}\log_2 5.
\end{align*}
Numerically: $\log_2 3 \approx 1.585$, $\log_2 5 \approx 2.322$, so
\[
  \E[\log_2 M] \approx 0.750 - 0.793 - 0.581 = -0.624.
\]

\paragraph{Size to first order in the exponent.}

By LLN, $\log_2 S_n \approx n\,\E[\log_2 M]$ with high probability as $n\to\infty$, i.e.,
\[
  S_n \;\doteq\; 2^{n\,\E[\log_2 M]}
  = \left(\frac{2}{3}\right)^{\!\frac{3n}{4}} \cdot \left(\frac{3}{5}\right)^{\!\frac{n}{4}},
\]
where $\doteq$ denotes equality to first order in the exponent.

\paragraph{Closed-form exponent.}
\begin{align*}
  S_n &\;\doteq\; \left[\left(\frac{2}{3}\right)^{\!3} \cdot \frac{3}{5}\right]^{\!n/4}
               = \left[\frac{8}{27}\cdot\frac{3}{5}\right]^{\!n/4}
               = \left(\frac{8}{45}\right)^{\!n/4}.
\end{align*}

\[
  \boxed{S_n \;\doteq\; \left(\frac{8}{45}\right)^{n/4} \approx (0.178)^{n/4} \approx 2^{-0.624\,n}.}
\]

\paragraph{Interpretation.}  Each cut reduces the cake to approximately
$2^{-0.624} \approx 0.649$ of its size.  After $n$ cuts the remaining piece is exponentially
small, decaying as $(0.649)^n$ to first order in the exponent.  This follows directly from
the AEP / LLN applied to the product of i.i.d.\ random scale factors.
